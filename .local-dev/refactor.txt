    on: m.company_id == c.id,
    where: m.user_id == ^user.id and c.status == "active",
    # ... same preload and order_by

  # ... nearly identical with different status filter

**Proposed Solution:**
Use a single base query with optional filters:

```elixir
defp user_companies_base_query(%User{} = user, opts \\ []) do
  query = from(c in Company,
    join: m in CompanyMembership,
    on: m.company_id == c.id,
    where: m.user_id == ^user.id,
    preload: [:memberships, :pilots, company_units: :master_unit],
    order_by: [desc: c.updated_at]
  )

  case Keyword.get(opts, :status) do
    nil -> query
    status when is_binary(status) -> where(query, [c], c.status == ^status)
    statuses when is_list(statuses) -> where(query, [c], c.status in ^statuses)
  end
end

def list_user_companies(%User{} = user), do: Repo.all(user_companies_base_query(user))
def list_user_active_companies(%User{} = user), do: Repo.all(user_companies_base_query(user, status: "active"))
def list_user_draft_companies(%User{} = user), do: Repo.all(user_companies_base_query(user, status: "draft"))
```

**Impact:** Reduces duplication from 3 full query definitions to 1 base + 3 one-liners. Easier to maintain and modify.

```elixir
defmodule AcesWeb.LiveHelpers do
  @moduledoc """
  Common patterns for LiveView components.
  """

  @doc """
  Subscribe to PubSub updates if socket is connected.

  ## Examples

      subscribe_if_connected(socket, fn ->
        Broadcasts.subscribe_company(company_id)
      end)
  """
  def subscribe_if_connected(socket, subscription_fn) do
    if Phoenix.LiveView.connected?(socket) do
      subscription_fn.()
    end
    socket
  end

  @doc """
  Standard reload pattern for entities.

  Reloads specified entities and updates socket assigns.
  """
  def reload_and_assign(socket, reloads) do
    assigns =
      Enum.reduce(reloads, %{}, fn {key, {module, func, args}}, acc ->
        value = apply(module, func, args)
        Map.put(acc, key, value)
      end)

    Phoenix.Component.assign(socket, assigns)
  end
end
```

Usage:

```elixir

    company: {Companies, :get_company!, [socket.assigns.company.id]},
    members: {Companies, :list_company_members, [socket.assigns.company]}

  Validates a condition and returns :ok or error tuple.

    if condition, do: :ok, else: {:error, error_message}

defmodule AcesWeb.LiveValidations do
  @doc """
  Validates socket state and returns :ok or {:error, message}.
  """
  def validate_permission(socket, key) do
    if socket.assigns[key], do: :ok, else: {:error, "Permission denied"}
  end

  def validate_status(socket, entity_key, expected_status) do
    actual = get_in(socket.assigns, [entity_key, :status])
    if actual == expected_status do
      :ok
    else
      {:error, "#{entity_key} must be in #{expected_status} state (currently #{actual})"}
    end
  end

  def validate_present(socket, key, error_message \\ nil) do
    message = error_message || "#{key} is required"
    case socket.assigns[key] do
      nil -> {:error, message}
      value -> {:ok, value}
    end
  end
end
```

**Impact:** More reusable validation helpers across LiveViews.

```elixir
defmodule AcesWeb.SortieLive.OmniVariantManager do
  @moduledoc """
  Handles OMNI variant reconfiguration logic for sortie deployment.
  """

  alias Aces.{Campaigns, Units}

  def load_variants(deployments) do
    deployments
    |> Enum.filter(&has_omni_unit?/1)
    |> Enum.map(fn d ->
      variants = Units.list_variants_for_chassis(d.company_unit.master_unit)
      {d.id, variants}
    end)
    |> Map.new()
  end

  def track_change(pending_changes, deployment_id, new_variant_id, current_variant_id) do
    if new_variant_id == current_variant_id do
      Map.delete(pending_changes, deployment_id)
    else
      Map.put(pending_changes, deployment_id, new_variant_id)
    end
  end

  def calculate_total_cost(sortie, pending_changes, omni_variants) do
    Campaigns.calculate_pending_refit_cost(sortie, pending_changes, omni_variants)
  end

  def calculate_effective_pv(sortie, pending_changes, omni_variants) do
    Campaigns.calculate_effective_deployed_pv(sortie, pending_changes, omni_variants)
  end

  defp has_omni_unit?(deployment) do
    deployment.company_unit &&
    deployment.company_unit.master_unit &&
    Units.is_omni?(deployment.company_unit.master_unit)
  end
end
```

**Impact:** Better separation of concerns, easier to test OMNI logic.
